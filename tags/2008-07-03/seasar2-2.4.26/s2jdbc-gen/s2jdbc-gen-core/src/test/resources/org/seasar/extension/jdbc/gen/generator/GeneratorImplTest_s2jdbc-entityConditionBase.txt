package bar;

import java.util.Date;
import org.seasar.extension.jdbc.where.ComplexWhere;
import org.seasar.extension.jdbc.where.condition.AbstractEntityCondition;
import org.seasar.extension.jdbc.where.condition.NotNullableCondition;
import org.seasar.extension.jdbc.where.condition.NotNullableStringCondition;
import org.seasar.extension.jdbc.where.condition.NullableCondition;
import org.seasar.extension.jdbc.where.condition.NullableStringCondition;

public abstract class AbstractFooCondition extends
        AbstractEntityCondition<AbstractFooCondition> {

    public AbstractFooCondition() {
    }

    public AbstractFooCondition(String prefix, ComplexWhere where) {
        super(prefix, where);
    }

    public NotNullableCondition<AbstractFooCondition, Integer> id =
        new NotNullableCondition<AbstractFooCondition, Integer>("id", this);

    public NotNullableStringCondition<AbstractFooCondition> name =
        new NotNullableStringCondition<AbstractFooCondition>("name", this);

    public NotNullableCondition<AbstractFooCondition, byte[]> lob =
        new NotNullableCondition<AbstractFooCondition, byte[]>("lob", this);

    public NullableCondition<AbstractFooCondition, Date> date =
        new NullableCondition<AbstractFooCondition, Date>("date", this);

    public NullableStringCondition<AbstractFooCondition> temp =
        new NullableStringCondition<AbstractFooCondition>("temp", this);

    public NotNullableCondition<AbstractFooCondition, Integer> version =
        new NotNullableCondition<AbstractFooCondition, Integer>("version", this);
}
